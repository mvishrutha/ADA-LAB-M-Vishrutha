#include <stdio.h>

#define MAX 100

void topologicalSort(int adj[MAX][MAX], int n) {
    int indegree[MAX] = {0};
    int queue[MAX];
    int front = 0, rear = -1;
    int i, j, count = 0;

    // Calculate indegrees
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            if (adj[i][j] == 1)
                indegree[j]++;
        }
    }

    // Insert vertices with indegree 0 into queue
    for (i = 0; i < n; i++) {
        if (indegree[i] == 0)
            queue[++rear] = i;
    }

    printf("Topological Ordering: ");

    while (front <= rear) {
        int v = queue[front++];
        printf("%d ", v);
        count++;

        for (j = 0; j < n; j++) {
            if (adj[v][j] == 1) {
                indegree[j]--;

                if (indegree[j] == 0)
                    queue[++rear] = j;
            }
        }
    }

    if (count != n)
        printf("\nGraph contains a cycle. Topological ordering not possible.");
}

int main() {
    int n, adj[MAX][MAX];
    int i, j;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            scanf("%d", &adj[i][j]);
        }
    }

    topologicalSort(adj, n);

    return 0;
}
